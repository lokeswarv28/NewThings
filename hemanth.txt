import pyspark.sql.functions as F

# Define a UDF to extract the minimum price from the price range
def extract_min_price(price_range):
    if '-' in price_range:
        min_price = price_range.split('-')[0].strip()
    elif 'Lakh' in price_range or 'Crore' in price_range:
        min_price = price_range.split()[0].strip()
    else:
        min_price = price_range
    return float(min_price.replace('Lakh', '').replace('Crore', '')) * (10**7 if 'Crore' in price_range else 10**5)

# Register the UDF
extract_min_price_udf = F.udf(extract_min_price, FloatType())
